class Solution {
    public ListNode deleteDuplicates(ListNode head) {
        if(head == null || head.next == null){
            return head;
        }
        ListNode left = head;
        ListNode right = head.next;

        while(right != null){
            if(left.val == right.val){
                right = right.next;
                left.next = right;
            }else{
                right = right.next;
                left = left.next;
            }
        }

        return head;
    }
}
