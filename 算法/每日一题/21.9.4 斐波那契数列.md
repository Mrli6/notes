class Solution {
    public int fib(int n) {
        int a = 0;
        int b = 1;

        while(n > 0){
            int temp = a%1000000007+b%1000000007;
            a = b%1000000007;
            b = temp%1000000007;
            n--;
        }

        return a;
    }
}
